---
title: Get Balance
api: GET /v1/custody/mpc/list_balances/ 
description: Obtain all balance information of the specified coin 
---
#### Request 

<ParamField body="coin" type="String" >name of the coin to be queried</ParamField> 
<ParamField body="chain_code" type="String" >Chain code</ParamField> 
<ParamField body="page_index" type="Int" required>page number</ParamField> 
<ParamField body="page_length" type="Int" required>number of entries per page; max: 50</ParamField> 

#### Response 

<ResponseField name="total" type="Int" >Total number of entries</ResponseField> 
<ResponseField name="address" type="String" >Address</ResponseField> 
<ResponseField name="coin" type="String" >Coin name (Cobo Custody)</ResponseField> 
<ResponseField name="chain_code" type="String" >Chain code</ResponseField> 
<ResponseField name="description" type="String" >Full name (reference only, subject to change)</ResponseField> 
<ResponseField name="balance" type="String" >Coin balance</ResponseField> 
<ResponseField name="decimal" type="Int" >Decimal precision</ResponseField> 
<ResponseField name="nft_code" type="String" >NFT code</ResponseField> 
<ResponseField name="token_id" type="String" >NFT token Id</ResponseField> 
<ResponseField name="contract_address" type="String" >NFT contract address</ResponseField>
