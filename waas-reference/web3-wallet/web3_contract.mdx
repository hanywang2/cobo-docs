---
#### Request 

<ParamField body="chain_code" type="String" required>Chain code</ParamField> 
<ParamField body="request_id" type="String" required>Withdrawal request ID (Universally unique ID for each user's withdraw request and less than or equal to 120 characters)</ParamField> 
<ParamField body="wallet_addr" type="String" required>Web3 wallet address</ParamField> 
<ParamField body="contract_addr" type="String" required>Contract address</ParamField> 
<ParamField body="method_id" type="String" required>Contract method id</ParamField> 
<ParamField body="method_name" type="String" required>Contract method name</ParamField> 
<ParamField body="args" type="String" required>Contract method args</ParamField> 
<ParamField body="gas_limit" type="Int" >Gas limit</ParamField> 

#### Response 

<ResponseField name="coin" type="String" >Coin Code</ResponseField>
